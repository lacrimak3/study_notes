# 1389번 케빈 베이컨의 6단계 법칙
>https://www.acmicpc.net/problem/1389

## 풀이

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int INF = 987654321;
int n, m;
int d[100][100];

int main(void) {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            i == j ? d[i][j] == 0 : d[i][j] = INF;
    while (m--) {
        int x, y, z = 1;
        cin >> x >> y;
        x--; y--;
        if (d[x][y] > z) {
            d[x][y] = z;
            d[y][x] = z;
        }
    }
    for (int k = 0; k < n; ++k) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (d[i][j] > d[i][k] + d[k][j]) {
                    d[i][j] = d[i][k] + d[k][j];
                }
            }
        }
    }
    int minV = INF, minU = 0;
    for (int i = 0; i < n; ++i) {
        int t = 0;
        for (int j = 0; j < n; ++j)
            t += d[i][j];
        if (minV > t) {
            minV = t;
            minU = i;
        }
    }
    cout << minU + 1;
    return 0;
}
```

친구관계이므로 양방향임을 주의
